git init // initialise git

git add . or filename // to stage file or all file

git commit -m "message" // to commit message

git checkout -b new-branch // to create new branch'

git checkout branch  // to switch to existing branch

git stash // to save uncommmited file. we mostly do while swich branches or while pulling from github

git push origin main // to push changes from local file to github

git pull origin main // to pull update from github to local file

git clone repo-name  // to clone repository
